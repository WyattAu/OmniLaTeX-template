    \begin{axis}[
        ybar,
        width=0.6\textwidth,
        height=5cm,
        xlabel={Year},
        ylabel={Revenue (\$M)},
        symbolic x coords={2020, 2021, 2022, 2023, 2024},
        xtick=data,
        ymin=0,

        nodes near coords align={vertical},
