\section*{Instructions}

This workbook accompanies the Advanced Calculus course (MATH 301). Complete
each exercise in the space provided. Show all work for full credit. You may
use a calculator, but all symbolic answers must be exact.

\section{Limits and Continuity}

\subsection{Exercises}

\begin{enumerate}[label=\textbf{1.\arabic*}]

    \item Evaluate the following limits. Show your work.

    \begin{enumerate}[label=(\alph*)]
        \item $\displaystyle\lim_{x \to 3} \frac{x^{2}-9}{x-3}$
        \vspace{2cm}

        \item $\displaystyle\lim_{x \to 0} \frac{\sin(5x)}{3x}$
        \vspace{2cm}

        \item $\displaystyle\lim_{x \to \infty} \frac{3x^{2}+2x}{5x^{2}-1}$
        \vspace{2cm}
    \end{enumerate}

    \item Determine whether the function is continuous at $x=2$. Justify
          your answer.

    \[
        f(x) = \begin{cases}
            x^{2}-1 & \text{if } x < 2 \\
            3       & \text{if } x = 2 \\
            2x+1    & \text{if } x > 2
        \end{cases}
    \]
    \vspace{3cm}

\end{enumerate}
