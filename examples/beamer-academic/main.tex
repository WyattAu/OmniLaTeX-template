\section{Motivation and Background}
%% ────────────────────────────────────────────────────

\begin{frame}{The Document Generation Problem}
  \begin{columns}[T]
    \column{0.48\textwidth}
    \begin{block}{Current State}
      \begin{itemize}
        \item Manual preamble configuration
        \item 50--200 lines of boilerplate
        \item Package conflicts common
        \item Theme changes require restarts
      \end{itemize}
    \end{block}

    \column{0.48\textwidth}
    \begin{block}{Desired State}
      \begin{itemize}
        \item One \texttt{\textbackslash documentclass} line
        \item 27 doctypes from one class
        \item 20+ institution configs
        \item Live theme switching
      \end{itemize}
    \end{block}
  \end{columns}

  \vspace{0.5em}
  \begin{alertblock}{Key Insight}
    Most document configuration is \emph{typographic knowledge} that can be encoded once and reused.
  \end{alertblock}
\end{frame}

\begin{frame}{Why LuaLaTeX?}
  \begin{enumerate}
    \item \textbf{Unicode native} --- Full CJK, Arabic, Hebrew, Cyrillic support without font encoding hacks.
    \item \textbf{OpenType fonts} --- Access to system fonts (Fira, Source Serif, Libertinus) with full OpenType features.
    \item \textbf{Lua scripting} --- Programmatic document generation, conditional includes, dynamic content.
    \item \textbf{Deterministic builds} --- Byte-for-byte reproducible PDFs via \texttt{SOURCE\_DATE\_EPOCH}.
  \end{enumerate}

  \begin{exampleblock}{TeX Live 2025+ Standard}
    LuaLaTeX is the default engine in TeX Live 2025. All 48 OmniLaTeX examples compile on LuaHBTeX 1.21.0+.
  \end{exampleblock}
\end{frame}

\section{Reproducibility}
%% ────────────────────────────────────────────────────

\begin{frame}{Deterministic Builds}
  \begin{columns}[T]
    \column{0.48\textwidth}
    \begin{block}{Mechanism}
      \begin{itemize}
        \item \texttt{SOURCE\_DATE\_EPOCH} pinned to Unix epoch (1700000000)
        \item PDF metadata timestamps derived from epoch
        \item LuaLaTeX ID suppression (\texttt{--safer --no-shell-escape})
        \item All external data hashed into cache keys
      \end{itemize}
    \end{block}

    \column{0.48\textwidth}
    \begin{block}{Verification}
      \begin{itemize}
        \item Two sequential builds produce byte-identical PDFs
        \item CI determinism check: file size within 3\% tolerance
        \item Visual regression with SSIM (Structural Similarity)
        \item Nix flake for hermetic build environment
      \end{itemize}
    \end{block}
  \end{columns}

  \vspace{1em}
  \[
    \text{Build}( \text{source}, t_1 ) \equiv \text{Build}( \text{source}, t_2 )
    \quad\forall\, t_1, t_2 \in \mathbb{T}
  \]
\end{frame}

\begin{frame}[fragile]{Nix and Docker Reproducibility}
  \begin{block}{Nix Flake (Linux/macOS)}
    \begin{semiverbatim}
\\# hermetic build with pinned TeX Live 2025
nix develop
python build.py build-example thesis
    \end{semiverbatim}
  \end{block}

  \begin{block}{Docker (Any Platform)}
    \begin{semiverbatim}
\\# multi-arch image (amd64 + arm64)
docker run -v \$(pwd):/workspace \\
  ghcr.io/wyattau/omnilatex-docker@sha256:dae352...
python build.py build-all
    \end{semiverbatim}
  \end{block}

  \begin{alertblock}{Supply Chain Security}
    Docker image digests are SHA-pinned across all CI configs. Digest changes require an automated PR with full CI validation.
  \end{alertblock}
\end{frame}
