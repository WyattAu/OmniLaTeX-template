\section{Introduction}
%% ────────────────────────────────────────────────────

\begin{frame}{Motivation: The Reproducibility Crisis}
  \begin{columns}[T]
    \column{0.5\textwidth}
    \begin{alertblock}{The Problem}
      \begin{itemize}
        \item 70\% of researchers failed to reproduce another scientist's experiments (Nature, 2016)
        \item Document generation is rarely treated as a reproducibility concern
        \item LaTeX documents depend on system state: fonts, packages, TeX version, timestamps
      \end{itemize}
    \end{alertblock}

    \column{0.5\textwidth}
    \begin{exampleblock}{Our Approach}
      \begin{itemize}
        \item Apply software engineering rigor to document generation
        \item Formal verification of template logic
        \item Property-based testing of output properties
        \item Deterministic, hermetic builds
      \end{itemize}
    \end{exampleblock}
  \end{columns}
\end{frame}

\begin{frame}{Research Questions}
  \begin{enumerate}
    \item Can formal methods (Lean 4) prove properties about LaTeX document template composition?
    \item Does property-based testing (Hypothesis) find bugs in template logic?
    \item Can deterministic PDF generation be achieved with LuaLaTeX?
    \item What architecture enables 27 document types from a single class without combinatorial explosion?
  \end{enumerate}
\end{frame}

\section{Background}
%% ────────────────────────────────────────────────────

\begin{frame}{LaTeX Document Classes: The Status Quo}
  \begin{block}{Current Ecosystem}
    \begin{itemize}
      \item \textbf{3 base classes:} article, report, book (or KOMA-Script equivalents)
      \item \textbf{300+ CTAN packages} needed to configure a typical thesis preamble
      \item \textbf{40+ single-university templates} on CTAN, each duplicating 90\% of configuration
      \item \textbf{No modularity:} changing doctype requires rewriting the entire preamble
    \end{itemize}
  \end{block}

  \vspace{1em}
  \begin{center}
    \textbf{Key insight:} The configuration space is combinatorial but the implementation space is not.\\
    27 doctypes $\times$ 20 institutions $\times$ 25 languages = 13,500 configurations,\\
    but only 31 modules, 27 doctype profiles, and one class file.
  \end{center}
\end{frame}

\begin{frame}{Lean 4 and Formal Verification}
  \begin{columns}[T]
    \column{0.48\textwidth}
    \begin{block}{Lean 4 Overview}
      \begin{itemize}
        \item Dependent type theory with inductive types
        \item Tactics for interactive and automated proving
        \item Lake build system for modular proof management
        \item Active community (2000+ GitHub stars)
      \end{itemize}
    \end{block}

    \column{0.48\textwidth}
    \begin{exampleblock}{What We Prove}
      \begin{itemize}
        \item Module load-order acyclicity
        \item Doctype option resolution determinism
        \item Institution config idempotence
        \item Language fallback chain well-foundedness
        \item Translation key coverage completeness
      \end{itemize}
    \end{exampleblock}
  \end{columns}
\end{frame}

\begin{frame}[fragile]{Property-Based Testing with Hypothesis}
  \begin{block}{Test Strategy}
    \begin{semiverbatim}
@given(doctype=sampled_from(DOCTYPES),
       language=sampled_from(LANGUAGES))
def test_combination_generates_pdf(doctype, language):
    tex = f"\\\\documentclass[doctype=doctype,
           language=language]\\{omnilatex\\}"
    result = compile_latex(tex)
    assert result.returncode == 0
    assert is_valid_pdf(result.pdf)
    \end{semiverbatim}
  \end{block}

  \begin{itemize}
    \item 75+ property-based tests exercise random combinations
    \item Shrinking: Hypothesis finds minimal failing examples automatically
    \item Edge cases: empty documents, 100+ page documents, deeply nested structures
  \end{itemize}
\end{frame}

    \begin{block}{Reproducibility}
      \begin{itemize}
        \item 48 examples: 100\% byte-identical on repeat builds
        \item File size tolerance: $<3\%$ across 5 consecutive builds
        \item PDF metadata: all timestamps derived from \texttt{SOURCE\_DATE\_EPOCH}
        \item Nix flake: hermetic TeX Live with exact package versions
      \end{itemize}
    \end{block}

    \column{0.5\textwidth}
    \begin{exampleblock}{Theorem (Reproducibility)}
      Let $\mathcal{B}(S, t)$ denote building source $S$ at wall-clock time $t$. Then:
      \[
        \forall S \in \text{OmniLaTeX}, \forall t_1, t_2 \in \mathbb{T}:
        \quad \text{SHA-256}(\mathcal{B}(S, t_1)) = \text{SHA-256}(\mathcal{B}(S, t_2))
      \]
      This holds under the condition that \texttt{SOURCE\_DATE\_EPOCH} is fixed and the TeX Live environment is hermetic (Nix or Docker).
    \end{exampleblock}
  \end{columns}
\end{frame}
