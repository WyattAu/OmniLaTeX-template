\chapter{Command Reference}

This appendix provides a comprehensive quick-reference table of every
OmniLaTeX custom command, organised by functional category.  Detailed usage
and examples for each command are found in the corresponding chapter.

\section{Document Setup}

\begin{longtblr}[
  caption = {Document setup commands},
  label = {tab:cmd-docsetup},
]{colspec = {l p{8.5cm}}, rowhead = 1}
  \toprule
  \textbf{Command} & \textbf{Description} \\
  \midrule
  \texttt{\textbackslash{}documenttype\{name\}} & Declare document type label in a profile
    \texttt{.sty} file \\
  \texttt{\textbackslash{}manualbrand\{label\}} & Insert manual-specific branding element \\
  \texttt{\textbackslash{}signaturefield\{width\}} & Underlined signature field for forms \\
  \bottomrule
\end{longtblr}

\section{Typography}

\begin{longtblr}[
  caption = {Typography and text formatting commands},
  label = {tab:cmd-typo},
]{colspec = {l p{8.5cm}}, rowhead = 1}
  \toprule
  \textbf{Command} & \textbf{Description} \\
  \midrule
  \texttt{\textbackslash{}enquote\{text\}} & Context-aware quotation marks via
    \ctanpackage{csquotes} \\
  \texttt{\textbackslash{}qq\{text\}} & Quick quotation shorthand: \texttt{\textbackslash quad}
    of text between equation parts \\
  \texttt{\textbackslash{}name\{x\}} & Format a proper noun or name (glossary-linked) \\
  \texttt{\textbackslash{}sym\{x\}} & Format a mathematical symbol name (glossary-linked) \\
  \texttt{\textbackslash{}sub\{x\}} & Format a subscript reference (glossary-linked) \\
  \texttt{\textbackslash{}cons\{x\}} & Format a constant or physical constant name \\
  \texttt{\textbackslash{}censor\{text\}} & Black-bar redaction preserving layout (requires
    \textsf{censoring} option) \\
  \texttt{\textbackslash{}hologo\{name\}} & Typeset a software logo (e.g.\ \LaTeX, \TeX) \\
  \texttt{\textbackslash{}boldPDFchapterstrue} & Enable bold chapter entries in PDF bookmarks \\
  \texttt{\textbackslash{}boldPDFchaptersfalse} & Disable bold chapter entries in PDF
    bookmarks \\
  \bottomrule
\end{longtblr}

\section{Mathematics}

\begin{longtblr}[
  caption = {Mathematics commands},
  label = {tab:cmd-math},
]{colspec = {l p{8.5cm}}, rowhead = 1}
  \toprule
  \textbf{Command} & \textbf{Description} \\
  \midrule
  \texttt{\textbackslash{}abs\{x\}} & Absolute value: $|x|$ (glossary-linked) \\
  \texttt{\textbackslash{}parens\{x\}} & Parenthesis delimiters: $(x)$; starred variant
    auto-scales \\
  \texttt{\textbackslash{}brackets\{x\}} & Bracket delimiters: $[x]$; starred variant
    auto-scales \\
  \texttt{\textbackslash{}braces\{x\}} & Brace delimiters: $\{x\}$; starred variant
    auto-scales \\
  \texttt{\textbackslash{}mean\{x\}} & Mean value: $\bar{x}$ (glossary-linked) \\
  \texttt{\textbackslash{}logmean\{x\}\{y\}} & Logarithmic mean: $(x - y)/\ln(x/y)$
    (glossary-linked) \\
  \texttt{\textbackslash{}vect\{x\}} & Vector notation: $\vec{x}$ (glossary-linked) \\
  \texttt{\textbackslash{}flow\{x\}} & Volumetric flow rate: $\dot{V}$ (glossary-linked) \\
  \texttt{\textbackslash{}difference\{x\}} & Delta/difference: $\Delta x$ (glossary-linked) \\
  \texttt{\textbackslash{}nablaoperator\{x\}} & Nabla operator applied: $\nabla x$
    (glossary-linked) \\
  \texttt{\textbackslash{}grad\{x\}} & Gradient operator: $\operatorname{grad} x$ \\
  \texttt{\textbackslash{}const\{x\}} & Mathematical constant (upright) \\
  \texttt{\textbackslash{}coloneq} & Colon-equals operator: $:=$ \\
  \texttt{\textbackslash{}deriv[n]\{f\}\{x\}} & Ordinary derivative:
    $\mathrm{d}^{n}f$ (glossary-linked) \\
  \texttt{\textbackslash{}deriv*[n]\{f\}\{x\}} & Partial derivative:
    $\partial^{n}f$ (glossary-linked) \\
  \texttt{\textbackslash{}fracderiv[n]\{f\}\{x\}} & Fractional ordinary derivative \\
  \texttt{\textbackslash{}fracderiv*[n]\{f\}\{x\}} & Fractional partial derivative \\
  \texttt{\textbackslash{}timederiv[n]\{f\}} & Time derivative:
    $\partial^{n}f / \partial t^{n}$ \\
  \texttt{\textbackslash{}timederiv*[n]\{f\}} & Partial time derivative (synonym) \\
  \texttt{\textbackslash{}posderiv[n]\{f\}} & Positional derivative:
    $\partial^{n}f / \partial x^{n}$ \\
  \texttt{\textbackslash{}posderiv*[n]\{f\}} & Partial positional derivative (synonym) \\
  \texttt{\textbackslash{}eqend} & Equation ending with period (aligned environments) \\
  \texttt{\textbackslash{}eqcomma} & Equation ending with comma (aligned environments) \\
  \texttt{\textbackslash{}circlednum\{n\}} & Circled number: \raisebox{0.3ex}{\tikz[baseline=(char.base)]{\node[circle,draw,inner sep=1pt,font=\small] (char) {1};}} \\
  \texttt{\textbackslash{}symbolplaceholder} & Placeholder for a symbol yet to be defined \\
  \bottomrule
\end{longtblr}
