%=============================================================================
\chapter{Introduction}
\label{sec:introduction}
%=============================================================================

This document presents the system design for DataPipe v4, a next-generation
streaming ETL platform. DataPipe v4 addresses the scalability and
operational limitations identified in the v3 architecture by introducing a
serverless execution model, distributed state management, and a unified
control plane.

\section{Design Goals}

\begin{enumerate}
    \item \textbf{Scale to 10M events/s} on a multi-region deployment.
    \item \textbf{Reduce operational overhead} through auto-scaling and
        self-healing.
    \item \textbf{Enable multi-tenancy} with resource isolation.
    \item \textbf{Improve developer experience} with a local development
        mode.
    \item \textbf{Maintain backward compatibility} with v3 pipeline
        definitions.
\end{enumerate}

\section{Throughput Calculation}

For a target of 10M events/s:

\begin{align}
    \text{Events per second} &= 10{,}000{,}000 \\
    \text{Avg.\ event size} &= 1{,}024\text{ bytes} \\
    \text{Ingress bandwidth} &= 10^7 \times 1024 = 10.24\text{ GB/s} \\
    \text{Egress bandwidth} &\approx 10.24\text{ GB/s (assuming 1:1 ratio)}
\end{align}

\section{Storage Requirements}

\begin{align}
    \text{Kafka retention} &= 7\text{ days} \\
    \text{Storage per day} &= 10.24\text{ GB/s} \times 86{,}400\text{ s} \approx 885\text{ TB} \\
    \text{Total Kafka storage} &= 885 \times 7 = 6{,}195\text{ TB} \\
    \text{Replication factor} &= 3 \\
    \text{Total disk} &= 6{,}195 \times 3 = 18{,}585\text{ TB} \approx 18.6\text{ PB}
\end{align}

\section{Pipeline Configuration Schema}

\begin{lstlisting}[language=yaml, caption={DataPipe v4 pipeline schema.}]
apiVersion: datapipe.io/v4
kind: Pipeline
metadata:
  name: user-events-to-warehouse
  tenant: acme-corp
  labels:
    team: data-engineering
    environment: production
spec:
  description: Ingest user events into the data warehouse
  deliveryGuarantee: exactly-once
  resources:
    cpu: 4
    memory: 8Gi
    maxConcurrency: 16
  stages:
    - name: source
      type: kafka
      config:
        topic: user-events
        consumerGroup: etl-warehouse
        bootstrapServers: kafka-0:9092,kafka-1:9092
    - name: filter
      type: filter
      config:
        predicate: "event_type IN ('click', 'purchase')"
    - name: enrich
      type: lookup
      config:
        source: user-profiles
        joinKey: user_id
        fields: [name, email, plan]
    - name: sink
      type: postgresql
      config:
        table: analytics.user_events
        batchSize: 10000
        flushInterval: 5s
\end{lstlisting}

\section{Metadata Tables}

\begin{lstlisting}[language=sql, caption={Core database schema.}]
CREATE TABLE pipelines (
    id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    tenant_id   UUID NOT NULL REFERENCES tenants(id),
    name        VARCHAR(255) NOT NULL,
    config      JSONB NOT NULL,
    status      VARCHAR(50) NOT NULL DEFAULT 'stopped',
    created_at  TIMESTAMPTZ NOT NULL DEFAULT now(),
    updated_at  TIMESTAMPTZ NOT NULL DEFAULT now(),
    UNIQUE(tenant_id, name)
);

CREATE TABLE pipeline_runs (
    id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    pipeline_id UUID NOT NULL REFERENCES pipelines(id),
    status      VARCHAR(50) NOT NULL,
    started_at  TIMESTAMPTZ,
    finished_at TIMESTAMPTZ,
    records_in  BIGINT DEFAULT 0,
    records_out BIGINT DEFAULT 0,
    errors      INTEGER DEFAULT 0
);

CREATE TABLE pipeline_checkpoints (
    id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    pipeline_id UUID NOT NULL REFERENCES pipelines(id),
    stage       VARCHAR(255) NOT NULL,
    offset      JSONB NOT NULL,
    created_at  TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE INDEX idx_pipelines_tenant ON pipelines(tenant_id);
CREATE INDEX idx_runs_pipeline ON pipeline_runs(pipeline_id);
CREATE INDEX idx_checkpoints_pipeline ON pipeline_checkpoints(pipeline_id);
\end{lstlisting}

\section{Multi-Region Deployment}

DataPipe v4 supports active-active multi-region deployment:

\begin{enumerate}
    \item Each region runs an independent Kafka cluster with cross-region
        mirroring via MirrorMaker 2.
    \item The control plane uses CockroachDB for globally consistent
        metadata.
    \item Pipelines are assigned to a primary region with automatic
        failover.
    \item Metrics are aggregated into a central monitoring cluster.
\end{enumerate}
