\section{Introduction}

This data dictionary defines the enterprise data model for the Meridian
Technologies platform. It provides a single source of truth for entity
definitions, attribute specifications, data type conventions, constraints,
and data ownership across all business domains. All teams must conform to
these definitions when designing new features or modifying existing schemas.

\section{Naming Conventions}

\begin{table}[h]
    \centering
    \begin{tabular}{ll}
        \toprule
        \textbf{Convention} & \textbf{Example} \\
        \midrule
        Entity names & PascalCase, singular noun (e.g., \texttt{CustomerOrder}) \\
        Attribute names & camelCase (e.g., \texttt{orderCreatedAt}) \\
        Table names & snake\_case, plural (e.g., \texttt{customer\_orders}) \\
        Primary keys & \texttt{\{entity\}\_id} (e.g., \texttt{customer\_order\_id}) \\
        Foreign keys & Referenced entity name + \texttt{\_id} \\
        Boolean columns & Prefixed with \texttt{is\_} or \texttt{has\_} \\
        Timestamps & Suffix \texttt{\_at} (e.g., \texttt{created\_at}) \\
        \bottomrule
    \end{tabular}
    \caption{Naming conventions for entities and attributes.}
\end{table}

\section{Data Type Reference}

\begin{table}[h]
    \centering
    \begin{tabular}{llp{5cm}}
        \toprule
        \textbf{Data Type} & \textbf{Storage} & \textbf{Notes} \\
        \midrule
        UUID & 16 bytes & RFC 4122 v4 random identifier \\
        VARCHAR(n) & Variable & Maximum n characters \\
        TEXT & Variable & Unlimited length text \\
        INTEGER & 4 bytes & Range: $-2^{31}$ to $2^{31}-1$ \\
        BIGINT & 8 bytes & Range: $-2^{63}$ to $2^{63}-1$ \\
        DECIMAL(p,s) & Variable & Exact numeric, precision p, scale s \\
        BOOLEAN & 1 byte & TRUE, FALSE, or NULL \\
        TIMESTAMP & 8 bytes & UTC with microsecond precision \\
        DATE & 3 bytes & Calendar date without time \\
        ENUM & 1-4 bytes & Named set of allowed values \\
        JSONB & Variable & Binary-serialized JSON document \\
        \bottomrule
    \end{tabular}
    \caption{Supported data types and storage characteristics.}
\end{table}
