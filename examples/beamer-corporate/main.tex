\section{Technical Deep Dive}
%% ────────────────────────────────────────────────────

\begin{frame}{Migration to OmniCloud v3}
  \begin{columns}[T]
    \column{0.55\textwidth}
    \textbf{Architecture Changes}
    \begin{itemize}
      \item Monolith $\rightarrow$ 12 microservices
      \item PostgreSQL $\rightarrow$ CockroachDB (multi-region)
      \item REST $\rightarrow$ gRPC for internal services
      \item CI/CD: 45 min $\rightarrow$ 12 min (incremental builds)
    \end{itemize}

    \column{0.45\textwidth}
    \textbf{Performance Impact}
    \begin{table}
      \begin{tabular}{l c c}
        \hline
        Metric    & Before & After \\
        \hline
        p50       & 120ms  & 45ms \\
        p99       & 850ms  & 200ms \\
        Throughput& 8k/s  & 22k/s \\
        \hline
      \end{tabular}
    \end{table}
  \end{columns}

  \vspace{1em}
  \begin{block}{Migration Strategy}
    Strangler Fig pattern: 12 weeks dual-write, 4 weeks read-migration, 2 weeks cutover. Zero-downtime deployment verified.
  \end{block}
\end{frame}

\begin{frame}[fragile]{Deployment Pipeline}
  \begin{block}{GitOps Workflow}
    \begin{semiverbatim}
main branch
  |-- CI: lint, test, build docker image
  |-- CD: deploy to staging (auto)
  |-- 15 min smoke test
  |-- Manual approval gate
  |-- CD: canary deploy to 5\% production
  |-- 30 min monitoring window
  |-- CD: rolling deploy to 100\%
    \end{semiverbatim}
  \end{block}

  \begin{itemize}
    \item All infrastructure defined in Terraform (version controlled)
    \item Secrets managed via HashiCorp Vault (no .env files)
    \item Observability: Prometheus + Grafana + OpenTelemetry traces
  \end{itemize}
\end{frame}
