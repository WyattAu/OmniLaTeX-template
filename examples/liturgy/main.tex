\section{Gathering}

\subsection{Prelude}
Instrumental music as the congregation gathers. \textit{Please observe a time of silent reflection.}

\subsection{Call to Worship}
\begin{quote}
Leader: The Lord is our light and our salvation.
\textbf{People: Whom shall we fear?}
Leader: The Lord is the stronghold of our lives.
\textbf{People: Of whom shall we be afraid?}
Leader: One thing we ask of the Lord.
\textbf{People: To dwell in the house of the Lord all the days of our lives.}
Leader: Come, let us worship the Lord.
\textbf{People: For the Lord is good; His steadfast love endures forever.}
\end{quote}

\subsection{Hymn of Praise}
Hymn 215: \textit{``Praise to the Lord, the Almighty''}
\textit{(LOBE DEN HERREN)}

\subsection{Prayer of Adoration}
\begin{quote}
Almighty God, Creator of heaven and earth, we gather in Your presence with gratitude and awe. You have called us from the noise of the world into the stillness of Your sanctuary. We praise You for the gift of life, for the beauty of creation, and for the grace that sustains us through every season. Open our hearts to receive Your Word. In Jesus' name we pray. Amen.
\end{quote}

\subsection{Prayer of Confession}
\begin{quote}
Holy God, we confess that we have not loved You with our whole heart. We have allowed distractions to pull us from Your presence. We have spoken harshly when gentleness was called for. We have remained silent when justice demanded our voice. We have hoarded what You have given us to share. Forgive us, Lord. Teach us Your ways and guide us in paths of righteousness, for the sake of Your Son, Jesus Christ. Amen.
\end{quote}

\subsection{Declaration of Pardon}
\begin{quote}
Leader: Hear the good news: ``If we confess our sins, God is faithful and just to forgive us our sins and to cleanse us from all unrighteousness'' (1 John 1:9).
\textbf{People: Thanks be to God!}
\end{quote}

\subsection{Passing of the Peace}
\begin{quote}
Leader: The peace of Christ be with you all.
\textbf{People: And also with you.}
\end{quote}
\textit{Please greet one another with a sign of peace.}

\section{Proclamation}

\subsection{Prayer for Illumination}
\begin{quote}
Lord, open our minds to understand Your Scriptures. May the words of my mouth and the meditation of all our hearts be acceptable in Your sight, O Lord, our Rock and our Redeemer. Amen.
\end{quote}

\subsection{Old Testament Reading}
Psalm 13 (read responsively)

\begin{quote}
Leader: How long, O Lord? Will you forget me forever?
\textbf{People: How long will you hide your face from me?}
Leader: How long must I wrestle with my thoughts and day after day have sorrow in my heart?
\textbf{People: How long will my enemy triumph over me?}
Leader: But I trust in your unfailing love; my heart rejoices in your salvation.
\textbf{People: I will sing the Lord's praise, for he has been good to me.}
\end{quote}

\subsection{New Testament Reading}
Romans 8:18--28 (NIV)

\textit{The congregation may be seated for the reading.}

\subsection{Gospel Reading}
Matthew 13:1--9, 18--23

\subsection{Sermon}

\textit{The Parable of the Sower} \\
\textbf{Text:} Matthew 13:1--9

\textit{(See attached sermon notes for full manuscript.)}

\subsection{Hymn of Response}
Hymn 420: \textit{``Be Thou My Vision''}
\textit{(SLANE)}

\section{Response}

\subsection{The Apostles' Creed}
\begin{quote}
I believe in God the Father Almighty, Creator of heaven and earth, and in Jesus Christ, His only Son, our Lord; who was conceived by the Holy Ghost, born of the Virgin Mary, suffered under Pontius Pilate, was crucified, dead, and buried; He descended into hell; the third day He rose again from the dead; He ascended into heaven, and sitteth on the right hand of God the Father Almighty; from thence He shall come to judge the quick and the dead.
\end{quote}

\subsection{Prayers of the People}
\textit{The congregation is invited to offer prayers aloud or in silence.}

\begin{quote}
For the church universal, that it may be a beacon of hope and a haven for the weary. \textbf{Lord, hear our prayer.}
\end{quote}

\begin{quote}
For the nations of the world, that leaders may seek justice, mercy, and peace. \textbf{Lord, hear our prayer.}
\end{quote}

\begin{quote}
For those who are suffering---the sick, the grieving, the lonely---that they may know the comfort of Your presence. \textbf{Lord, hear our prayer.}
\end{quote}

\begin{quote}
For this congregation, that we may grow in faith, deepen our love, and serve our neighbors with glad hearts. \textbf{Lord, hear our prayer.}
\end{quote}

\subsection{Offering}
\textit{An offering will be collected. You may also give online at the church website.}

\subsection{Doxology}
\begin{quote}
Praise God from whom all blessings flow;
Praise Him, all creatures here below;
Praise Him above, ye heavenly host;
Praise Father, Son, and Holy Ghost. Amen.
\end{quote}
